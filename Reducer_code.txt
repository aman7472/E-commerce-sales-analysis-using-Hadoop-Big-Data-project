#!/usr/bin/env python
import sys

current = ""
sales = profit = count = 0

for line in sys.stdin:
    cat, val = line.strip().split('\t')
    s, p = val.split(',')

    s = float(s)
    p = float(p)

    if cat == current:
        sales += s
        profit += p
        count += 1
    else:
	if current:
            print "%s\t%s\t%s\t%s\t%s" % (current, sales, profit, sales/count, profit/count)
        current = cat
        sales = s
        profit = p
        count = 1

if current:
    print "%s\t%s\t%s\t%s\t%s" % (current, sales, profit, sales/count, profit/count)
